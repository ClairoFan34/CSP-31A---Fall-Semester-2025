// This program prints "You Pass" if a student's average is
// 60 or higher and prints "You Fail" otherwise
// Angel Perez
#include <iostream>
using namespace std;
int main()
{
	float average; // holds the grade average
	cout << "Input your average:" << endl;
	cin >> average;
	if (average > 100 || average < 0 )
	    cout << "Invalid input";
	else if (average >= 90)
	    cout << "You passed with an A";
	else if (average >= 80)
	    cout << "You passed with a B";
	else if (average >= 60)
	    cout << "You pass";
	else 
	    cout << "You fail";
	return 0;
}
